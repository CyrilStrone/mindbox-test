import '../styles/todo-new.css'
 
import { SetStateAction, useState } from 'react';
 
import { TodoNewProps, TodoValueProps } from '../organelles/todo'
 
export const addNewItem = (newText: string, value: TodoValueProps[] | null, setValue: React.Dispatch<React.SetStateAction<TodoValueProps[] | null>>, setNewText: React.Dispatch<SetStateAction<string>>) => {
  if (!newText) return
  const newItem = {
    text: newText,
    isCheck: false,
    date: new Date(),
  };
 
  if (value) {
    setValue([...value, newItem]);
  } else {
    setValue([newItem]);
  }
  setNewText("");
};
 
export function TodoNew(props: TodoNewProps) {
